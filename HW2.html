# (d)
num_simulations = 1000
beta_1_est = rep(NA, num_simulations)
for (i in 1:num_simulations) {
  error = rnorm(100,0,1)
  Y = 2 + 3 * X1 + 5 * log(X2) + error
  model = lm(Y~X1 + log(X2))
  beta_1_est[i] = model$coefficients[[2]]
}
beta_1_bias = mean(beta_1_est) - 3
beta_1_bias
